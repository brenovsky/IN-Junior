Desafio 6

Peça ao usuário um nome e uma idade. Após isso, pergunte se ele quer inserir mais cadastros. Se sim, repita até não querer mais. Quando encerrar, mostre no console todos os cadastros realizados e a quantidade de cadastros feitos.

Resposta:

        
            var nomes = []
            var idades = []

            while (true) {
                let nome = prompt("Qual é o nome?")
                nomes.push(nome)

                let idade = prompt("Qual é a idade?")
                idades.push(idade)

                let stop = prompt("Continuar? S/N")

                if (stop == "N" || stop == "n") {
                    break
                }
            }

            for (let i = 0; i < nomes.length; i ++) {
                console.log(nomes[i] + " tem " + idades[i] + " anos.")
            }

            console.log("Foram feitos " + nomes.length + " cadastros.")
